Expand description

Takes an (ideally u32) ChainIdentifier and a chain type and re-computes an updated chain id with the chain type prepended to it. The resulting chain id is 6 bytes long and so requires a u64 to represent it.

use webb_primitives::utils::compute_chain_id_type;
pub type ChainId = u64;
let chain_id: u32 = 5;
let chain_type: [u8; 2] = [2, 0];

let chain_id_type: ChainId = compute_chain_id_type::<ChainId>(chain_id.into(), chain_type);
